    <script>

        // ★ 이 부분을 선생님의 데이터로 수정하세요 ★

        const flashcards = [

            { img: "https://drive.google.com/file/d/1LYJdcHgFEsvjg6en8aFMTKYajzOuqtSV/view?usp=drive_link", text: “여우" },

            { img: "https://drive.google.com/file/d/1LYJdcHgFEsvjg6en8aFMTKYajzOuqtSV/view?usp=drive_link", text: “아이" },

            { img: "https://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “오이” },

     { img: "ahttps://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “이유” },

            { img: "https://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “우유" },

            { img: "https://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “이” },

     { img: "https://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “오” },

            { img: "https://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “아이" },

            { img: "https://drive.google.com/file/d/1u0gmUuwL0FJPcNNISeXLymyRFr4mVr7J/view?usp=drive_link", text: “요” },


            // 필요한 만큼 계속 추가하세요 { img: "파일명", text: "단어" }

        ];


        let currentIndex = 0;


        function showCard() {

            const card = flashcards[currentIndex];

            document.getElementById('card-img').src = card.img;

            document.getElementById('card-text').innerText = card.text;

            document.getElementById('progress').innerText = `${currentIndex + 1} / ${flashcards.length}`;

            

            // 한국어 음성(TTS) 재생

            speak(card.text);

        }


        function speak(msg) {

            // 재생 중인 음성 정지

            window.speechSynthesis.cancel();

            const utterance = new SpeechSynthesisUtterance(msg);

            utterance.lang = 'ko-KR';

            utterance.rate = 1.0; 

            window.speechSynthesis.speak(utterance);

        }


        function nextCard() {

            currentIndex++;

            if (currentIndex >= flashcards.length) {

                currentIndex = 0; // 무한 루프: 마지막이면 처음으로

            }

            showCard();

        }


        // 첫 카드 실행 (사용자 상호작용 후 재생을 권장하므로 처음엔 텍스트만 표시)

        window.onload = () => {

            document.getElementById('progress').innerText = `총 ${flashcards.length}장의 카드가 있습니다`;

        };
