<script>

const flashcards = [

    { img: "https://raw.githubusercontent.com/wakuseiboy-prog/korean/main/01-1.png", text: “안녕하십니까? 저는 다나카 입니다." },

    { img: "https://raw.githubusercontent.com/wakuseiboy-prog/korean/main/02-2.png", text: “안녕하세요. 이유진 입니다. 잘 부탁합니다." },

    { img: "https://raw.githubusercontent.com/wakuseiboy-prog/korean/main/03-3.png", text: “다나카 씨는 대학생 입니까?" },

    { img: "https://raw.githubusercontent.com/wakuseiboy-prog/korean/main/04-4.png", text: “예, 대학생입니다." },


];


let currentIndex = 0;

let showingAnswer = false;


// 질문만 표시

function showQuestion() {

    const card = flashcards[currentIndex];

    document.getElementById('card-img').src = card.img;

    document.getElementById('card-text').innerText = "";

    document.getElementById('progress').innerText = `${currentIndex + 1} / ${flashcards.length}`;

}


// 정답 표시 + TTS (0.5초 딜레이)

function showAnswer() {

    const card = flashcards[currentIndex];

    document.getElementById('card-text').innerText = card.text;

    

    // 0.5초 지연 후 발음

    setTimeout(() => {

        const utterance = new SpeechSynthesisUtterance(card.text);

        utterance.lang = 'ko-KR';

        utterance.rate = 0.9; // 속도 조절

        window.speechSynthesis.cancel();

        window.speechSynthesis.speak(utterance);

    }, 500);

}


// 다음 카드

function nextCard() {

    if (!showingAnswer) {

        showAnswer();

        showingAnswer = true;

    } else {

        currentIndex++;

        if (currentIndex >= flashcards.length) currentIndex = 0;

        showingAnswer = false;

        showQuestion();

    }

}


// 메뉴로 돌아가기

function goMenu() {

    window.location.href = "index.html";

}


// 처음 로드

window.onload = () => {

    showQuestion();

};

</script>
